237 uint32_t core_exception_handler(unsigned long mcause, unsigned long sp)
238 {
239  uint32_t EXCn = (uint32_t)(mcause & 0X00000fff);
240  EXC_HANDLER exc_handler;
241 
242  if ((EXCn < MAX_SYSTEM_EXCEPTION_NUM) && (EXCn >= 0)) {
243  exc_handler = (EXC_HANDLER)SystemExceptionHandlers[EXCn];
244  } else if (EXCn == NMI_EXCn) {
245  exc_handler = (EXC_HANDLER)SystemExceptionHandlers[MAX_SYSTEM_EXCEPTION_NUM];
246  } else {
247  exc_handler = (EXC_HANDLER)system_default_exception_handler;
248  }
249  if (exc_handler != NULL) {
250  exc_handler(mcause, sp);
251  }
252  return 0;
253 }

287 {
288  if ((IRQn > SOC_INT_MAX) || (shv > ECLIC_VECTOR_INTERRUPT) \
289  || (trig_mode > ECLIC_NEGTIVE_EDGE_TRIGGER ) || (handler == NULL)) {
290  return -1;

293  /* set interrupt vector mode */

296  ECLIC_SetTrigIRQ(IRQn, trig_mode);
297  /* set interrupt level */
298  ECLIC_SetLevelIRQ(IRQn, lvl);
299  /* set interrupt priority */
300  ECLIC_SetPriorityIRQ(IRQn, priority);
301  /* set interrupt handler entry to vector table */
302  ECLIC_SetVector(IRQn, (rv_csr_t)handler);
303  /* enable interrupt */
304  ECLIC_EnableIRQ(IRQn);
